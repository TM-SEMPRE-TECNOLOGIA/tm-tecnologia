3

4

Código de exemplo (Playwright):

// frontend/e2e/fluxo-medidas.spec.ts
import { test, expect } from '@playwright/test';

test('T2: Preenche medidas e vê total calculado', async ({ page }) => {
  await page.goto('http://localhost:3000');
  await page.click('text=2626');        // seleciona contrato
  await page.click('button:has-text("Próximo")'); // Step 2
  
  // Seleciona fotos
  const fc = await page.waitForEvent('filechooser');
  await page.click('button:has-text("Selecionar Pasta")');
  await fc.setFiles(['./e2e/fixtures/foto-01.jpg', './e2e/fixtures/foto-02.jpg']);
  
  await expect(page.locator('.block-card')).toHaveCount(2);
  
  // Seleciona item e preenche medidas
  await page.locator('.block-card').first().locator('select').selectOption('17.6');
  await page.locator('.mf-input').first().fill('3.50');  // largura
  await page.locator('.mf-input').nth(1).fill('2.80');   // altura
  
  // Total: 3.50 × 2.80 × 1 face = 9.80 m²
  await expect(page.locator('.block-total-pill').first()).toContainText('9,80');
});

test('T4: Geração com download', async ({ page }) => {
  // Mock da API
  await page.route('**/generate-v2', route => route.fulfill({
    status: 200,
    contentType: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
    body: Buffer.from('fake-docx'),
  }));
  
  // ... setup ...
  
  const [download] = await Promise.all([
    page.waitForEvent('download'),
    page.click('button:has-text("Gerar .docx")'),
  ]);
  
  expect(download.suggestedFilename()).toMatch(/Relatorio-OS1753.*\.docx/);
});

test('T5: Validação — bloqueia geração sem medidas', async ({ page }) => {
  await page.goto('http://localhost:3000');
  // ... vai para Step 3 sem preencher ...
  
  await expect(page.locator('button:has-text("Gerar .docx")')).toBeDisabled();
  await expect(page.locator('.toast')).toContainText('Preencha as medidas');
});
